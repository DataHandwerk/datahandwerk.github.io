 @Parameter_name VARCHAR(100)

UPDATE p
SET [Parameter_value] = @Parameter_value
FROM [repo].[Parameter] [p]
WHERE [p].[Parameter_name] = @Parameter_name
 AND [p].[sub_Parameter] = @sub_Parameter
